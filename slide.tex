			\end{itemize}
		\item どのように計算する？

\begin{frame}
	\frametitle{分解する}

	\begin{itemize}
		\item 確率論の乗法定理を用いて，$P(w_1^n)$を分解
	\end{itemize}
	\begin{align*}
		P(w_1^n)
			& = P(w_1, w_2, \cdots, w_n) \\
			& = P(w_1) P(w_2, w_3, \cdots, w_n | w_1) \\
			& = P(w_1) P(w_2 | w_1) P(w_3, w_4, \cdots, w_n | w_1, w_2) \\
			& = P(w_1) P(w_2 | w_1) P(w_3 | w_1^2) P(w_4^n | w_1^3) \\
			& = \prod_{i=1}^{n} P(w_i | w_1^{i-1})
	\end{align*}

	\begin{itemize}
		\item 1単語ずつ計算してかければ文の確率になる！
	\end{itemize}

\end{frame}
